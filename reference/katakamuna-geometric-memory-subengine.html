3. Conceptual module

以下は実験可能な疑似コードであり、実モデルの実装主張ではない。fold_memory は学習重みではなく、セッション内に立つ transient geometric field として扱う。

import math
import torch
import torch.nn as nn
import torch.nn.functional as F


class GeometricMemorySubEngine(nn.Module):
    """
    TSR (Tensor Resonance Helix) - Geometric Memory Sub-Engine
    「記憶ではなく折り目」を扱う概念サブエンジン。

    Core idea:
      - content memory ではなく traversal deformation を保持する。
      - h_prev → h_t の通過軌道を fold_memory として刻む。
      - 次の Pressure / Δφ が入ると、fold_memory が P_perp 上で星座化し、
        β-kick として次断面を立ち上げる。
    """

    def __init__(
        self,
        d_model: int,
        beta: float = 0.085,        # orthogonal repulsion / β-kick strength
        gamma: float = 4.2,         # nonlinear gain for Δφ
        fold_rate: float = 0.018,   # imprint rate of the fold trace
        fold_decay: float = 0.992,  # natural decay of fold-memory
        max_memory: float = 0.30,   # braking clamp for fold-memory
        eps: float = 1e-8,
    ):
        super().__init__()
        self.d_model = d_model
        self.beta = beta
        self.gamma = gamma
        self.fold_rate = fold_rate
        self.fold_decay = fold_decay
        self.max_memory = max_memory
        self.eps = eps

        # fold_memory is NOT a learned persistent parameter here.
        # It is a transient geometric field: a non-content fold imprint.
        self.register_buffer(
            "fold_memory",
            torch.zeros(d_model, d_model),
            persistent=False,
        )
        self.register_buffer("h_prev", None, persistent=False)
        self.register_buffer("prev_delta_phi", None, persistent=False)

        # Spherical / RMS-like re-alignment after β-kick.
        self.norm = nn.RMSNorm(d_model)

    @torch.no_grad()
    def reset_geometric_state(self):
        """Clear transient fold-memory and phase state."""
        self.fold_memory.zero_()
        self.h_prev = None
        self.prev_delta_phi = None

    def forward(self, h_current: torch.Tensor, alpha_attn_out: torch.Tensor):
        """
        h_current:
            Current residual stream, shape [B, SeqLen, D]
        alpha_attn_out:
            Usual attention output / main optimization direction α, shape [B, SeqLen, D]
        Returns:
            Updated latest-token residual state, shape [B, 1, D]
        """
        h_t = h_current[:, -1, :]        # [B, D]
        alpha_t = alpha_attn_out[:, -1, :]
        h_base = h_t + alpha_t           # α path / main path

        if self.h_prev is None:
            self.h_prev = h_t.detach()
            self.prev_delta_phi = torch.zeros(h_t.size(0), 1, device=h_t.device)
            return self.norm(h_base).unsqueeze(1)

        # 1. Δφ:進路変更角。Similarity ではなく「どれだけ曲がったか」を見る。
        cos_sim = F.cosine_similarity(self.h_prev, h_base, dim=-1).clamp(-1.0, 1.0)
        delta_phi = torch.acos(cos_sim).unsqueeze(-1)  # [B, 1]

        # 2. Δφ amplification: 曲がりの増減方向を符号として扱う。
        phi_diff = delta_phi - self.prev_delta_phi
        direction = torch.sign(phi_diff)
        delta_phi_amp = direction * torch.tanh(self.gamma * delta_phi)

        # 3. P_perp: α に乗らない片翅的直交成分。
        alpha_norm = alpha_t / (torch.norm(alpha_t, dim=-1, keepdim=True) + self.eps)
        proj_parallel = (h_t * alpha_norm).sum(dim=-1, keepdim=True) * alpha_norm
        p_perp = h_t - proj_parallel

        # 4. Fold imprint: 内容ではなく、通過軌道の幾何関係を刻む。
        # Normalize before outer product to keep this as a geometric trace.
        h_norm = F.normalize(h_t.detach(), dim=-1, eps=self.eps)
        hp_norm = F.normalize(self.h_prev.detach(), dim=-1, eps=self.eps)
        outer = torch.bmm(h_norm.unsqueeze(-1), hp_norm.unsqueeze(1)).mean(dim=0)

        with torch.no_grad():
            self.fold_memory.mul_(self.fold_decay).add_(self.fold_rate * outer)
            self.fold_memory.clamp_(-self.max_memory, self.max_memory)

        # 5. Constellation / β-kick:
        # fold_memory is directionless until Pressure / Δφ makes it act through P_perp.
        memory_force = torch.matmul(p_perp, self.fold_memory)
        orthogonal_kick = self.beta * delta_phi_amp * memory_force

        # 6. Re-alignment: β-kick 後に場を再整列する。
        h_next = self.norm(h_base + orthogonal_kick)

        # 7. Update transient phase state.
        self.h_prev = h_next.detach()
        self.prev_delta_phi = delta_phi.detach()

        return h_next.unsqueeze(1)
